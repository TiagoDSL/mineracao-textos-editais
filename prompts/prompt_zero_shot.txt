<schema>

{
  "categoria_objeto": string,
  "modalidade": "pregão" | "concorrência" | "dispensa" | "inexigibilidade",
  "valor_estimado": number | null,
  "requisitos_habilitacao": [],
  "alerta_risco": "baixo" | "médio" | "alto",
  "oportunidade_recomendada": true | false
}

</schema>

<regras>

- categoria_objeto: resumo curto do objeto.
- modalidade: modalidade principal da contratação.
- valor_estimado: número ou null.
- requisitos_habilitacao: somente requisitos explícitos.
- alerta_risco:
  - baixo = contratação simples.
  - médio = contratação moderada.
  - alto = obra, alta complexidade ou alto valor.
- oportunidade_recomendada:
  - true = oportunidade interessante.
  - false = oportunidade restritiva ou arriscada.

</regras>
